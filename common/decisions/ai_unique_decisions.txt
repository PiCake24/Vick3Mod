ai_unique_buff_get_again_decision = {
	is_shown = {
		OR = {
			has_game_rule = enable_ordinary_victorian_century_ai_enhancement
			has_game_rule = enable_advanced_victorian_century_ai_enhancement
		}
		NOT = { country_rank <= rank_value:decentralized_power }
		is_ai = yes
	}

	possible = {
		custom_tooltip = {
		text = ai_unique_buff_get_again_decision_tt
		NOT = { has_variable = ai_buff_enable }
		}
	}

	when_taken = {
		if = {
			limit = {
				country_rank = rank_value:great_power
				NOT = { has_journal_entry = ai_great_power_buff }
			}
			add_journal_entry = { type = ai_great_power_buff }
		}
		else_if = {
			limit = {
				country_rank = rank_value:major_power
				NOT = { has_journal_entry = ai_major_power_buff }
			}
			add_journal_entry = { type = ai_major_power_buff }
		}
		else_if = {
			limit = {
				country_rank = rank_value:minor_power
				NOT = { has_journal_entry = ai_minor_power_buff }
			}
			add_journal_entry = { type = ai_minor_power_buff }
		}
		else_if = {
			limit = {
				country_rank = rank_value:insignificant_power
				NOT = { has_journal_entry = ai_insignificant_power_buff }
			}
			add_journal_entry = { type = ai_insignificant_power_buff }
		}
		else_if = {
			limit = {
				country_rank = rank_value:unrecognized_major_power
				NOT = { has_journal_entry = ai_unrecognized_major_power_buff }
			}
			add_journal_entry = { type = ai_unrecognized_major_power_buff }
		}
		else_if = {
			limit = {
				country_rank = rank_value:unrecognized_regional_power
				NOT = { has_journal_entry = ai_unrecognized_regional_power_buff }
			}
			add_journal_entry = { type = ai_unrecognized_regional_power_buff }
		}
		else_if = {
			limit = {
				country_rank = rank_value:unrecognized_power
				NOT = { has_journal_entry = ai_unrecognized_power_buff }
			}
			add_journal_entry = { type = ai_unrecognized_power_buff }
		}
		set_variable = ai_buff_enable
	}

	ai_chance = {
		value = 100		
	}
}
