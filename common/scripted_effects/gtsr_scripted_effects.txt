update_gtsr_pop_culture = {
    every_scope_pop = {
        limit = {
            pop_is_gts_job_trigger = yes
            NOT = { culture = cu:gts_cult1 }
        }
        change_pop_culture = {
            target = cu:gts_cult1
            value = 1
        }
#        if = {
#            limit = {
#                is_pop_type = wsg_cache
#            }
#            change_poptype = pop_type:wsg_bureaucrats
#        }
    }
    every_scope_pop = {
        limit = {
            culture = cu:gts_cult1
            religion != owner.religion
            owner = {
                country_average_religious_acceptance = {
                    target = prev.religion
                    value < 15
                }
            }
        }
        change_pop_religion = {
            target = owner.religion
            value = 1
        }
    }
}

gtsr_pop_startup_effect = {
    hidden_effect = {
        every_scope_state = {
            limit = {
                is_incorporated = yes
                is_sea_adjacent = yes
            }
            every_scope_pop = {
                limit = {
                    NOR = {
                        strata = lower
                        culture = cu:gts_cult1
                        is_pop_type = farmers
                        AND = {
                            is_pop_type = aristocrats
                            owner = {
                                has_law_or_variant = law_type:law_serfdom
                            }
                        }
                    }
                }
                change_pop_culture = {
                    target = cu:gts_cult1
                    value = 0.1
                }
            }
        }
    }
}
